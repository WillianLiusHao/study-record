### 二叉树  
  - 遍历 
    + 前中后序 
    + 层次遍历
      + 队列思想
      + 记录每层数目
      + 数组包裹每层数据，进入下一层
      ```js
        const levelOrder = (root)=> {
          if(!root) return root
          /**
           * cur: 当前层数目
          * next: 下一层数目
          * path: 当前层数据
          * queue: 存数队列
          */
          let cur = 1,next = 0, path = [],queue = [root], res=[];
          while(queue.length > 0) {
            const curNode = queue.shift()
            path.push(curNode.val)
            if(curNode.left) {
              queue.push(curNode.left)
              next++
            }
            if(curNode.right) {
              queue.push(curNode.right)
              next++
            }
            cur--
            if(cur ===0) {
              // 进入下一层
              cur = next
              next = 0
              // 每一层数据（path）用数组包起来
              res.push(path)
              path = []
            }
          }
          return res
        }
      ```
